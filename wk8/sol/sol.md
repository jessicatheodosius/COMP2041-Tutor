### Question 3

```
sub nlines {
    my ($n) = @_;
    my $text = "";
    $text .= <> foreach (1..$n);
    return $text;
}

$n = shift @ARGV or die;
print nlines($n);
```

### Question 4

```
$fn = shift @ARGV;

foreach $f (glob '*.c') {
	open F, "<$f";
	while ($line = <F>) {
		if ($line =~ /^\s.*\b$fn\s*\(.*?\)/) {
			printf "%s:%d function %s used\n", $f, $., $fn;
		} else if ($line =~ /^\S.*\b$fn\s*\(.*?\);/) {
			printf "%s:%d function %s declared\n", $f, $., $fn;
		} else if ($line =~ /^\S.*\b$fn\s*\(.*?\)\s*{/) {
			printf "%s:%d function %s defined\n", $f, $., $fn;
		}
	}
	close F;
}

```

### Question 6

```
sub include {
	my ($f) = @_;
    open (my $F, "<$f") or die;

	while (my $line = <$F>) {
		if ($line =~ /^#include "(.*)"$/) {
			include($1);
		} else {
			print $line;
		}
	}
}

include ($ARGV[0]);
```
